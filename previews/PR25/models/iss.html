using ReachabilityAnalysis, MAT

X = Universe(270)

input domain

U = Hyperrectangle([0.05, 0.9, 0.95], [0.05, 0.1, 0.05])

function model(X0)
    S = @system(x' = Ax + Bu, x ∈ X, u ∈ U)
    return IVP(S, X0)
end
